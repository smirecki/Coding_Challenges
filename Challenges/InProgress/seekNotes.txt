                    str = str.toUpperCase();
                    rollString = rollString + str;
                    counter++;
                } else {
                    str = str.toLowerCase(); // make lower
                    rollString = rollString + str;
                    counter++;
                }
            } else {
                rollString = rollString + str; // not letter
